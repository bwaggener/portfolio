episode_metrics %>%
  group_by(season) %>%
  summarise(average_qr = mean(question_ratio, na.rm = TRUE)) %>%
  ggplot(aes
         (x = season, y = average_qr)) +
  geom_col(fill = "#B28142") +
  labs(title = "Average Question Ratio per Season",
       x = "Season",
       y = "Average question ratio") +
  theme_classic()
